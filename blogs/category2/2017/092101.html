仿京东显示密码
```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <style>
    .box{
      width: 400px;
      border-bottom:1px solid red;
      margin: 100px auto;
      position: relative;
    }
    .box input{
      width:370px;
      height: 30px;
      outline: none;
      border: 0;
      /* boder可以来消除当前input框中的边框 */
    }
    .box button{
      position: absolute;
      top: 0;
      right: 0;

    }
  </style>
</head>
<body>
  <!-- 核心：点击的时候，将密码框类型改为一个文本框，就可以看见里面的密码内容了 -->
  <div class="box">
    <label for=""></label>
    <input type="password" id="eye">
    <button type="button" id="pwd">点击</button>
  </div>
  <script>
    //获取元素
    var eye=document.getElementById('eye');
    var pwd=document.getElementById('pwd');
    var flag=0;
    pwd.onclick=function(){
      //一个按钮可以进行点击多此的时候，就需要进行设置当前的flag属性
      if(flag===0){
        eye.type='text';
        flag=1;//进行赋值操作
      }else{
        eye.type="password";
        flag=0;
      }

    }

  </script>
</body>
</html>

```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">

      height: 400px;

      margin: 100px auto;
      position: relative;

      width: 380px;
      height: 380px;
      background-color: pink;
      margin: 10px;
    }
    .close-btn{
      font-size: 12px;
      width: 10px;
      height: 10px;
      padding: 0;
      line-height: 10px;
      text-align: center;
      border: 1px solid red;
      position: absolute;
      top: 0;
      left: 0;
      cursor: pointer;
    }
  </style>
</head>
<body>

  <div class="box"></div>
  <i class="close-btn">×</i>
</div>
<script>
  var bar=document.querySelector('.box-bar');
  var btn=document.getElementsByClassName('close-btn')[0];//区别出className的用法实例

  btn.onclick=function(){
    bar.style.display='none'
  }
</script>
</body>
</html>
